Validation de données avec Pydantic

Guide complet sur la validation et la sérialisation de données avec Pydantic : BaseModel, Field, validators personnalisés, ConfigDict, dataclasses et gestion des erreurs de validation

Dernière mise à jour : 12.01.2026

Introduction à Pydantic

Pydantic est une bibliothèque Python de validation de données utilisant les annotations de type. Elle permet de définir des modèles de données avec une validation automatique, une sérialisation et une gestion des erreurs robuste.

Installation

pip install pydantic

Import de base

from pydantic import BaseModel, Field, field_validator, ValidationError
from pydantic.dataclasses import dataclass
from typing import Optional, List
from datetime import datetime

BaseModel - Les fondamentaux

BaseModel est la classe de base de Pydantic pour créer des modèles de données avec validation automatique.

Modèle simple

from pydantic import BaseModel

class User(BaseModel):
    id: int
    name: str
    email: str
    age: int
    is_active: bool = True

# Création d'une instance
user = User(
    id=1,
    name="Alice",
    email="alice@example.com",
    age=25
)

print(user.name)  # Alice
print(user.model_dump())  # Convertir en dictionnaire

Validation automatique des types

# Conversion automatique des types
user = User(
    id="123",  # Sera converti en int
    name="Bob",
    email="bob@example.com",
    age="30"   # Sera converti en int
)

print(user.id)   # 123 (int)
print(user.age)  # 30 (int)

Gestion des erreurs avec ValidationError

from pydantic import ValidationError

try:
    user = User(
        id="invalid",  # Ne peut pas être converti en int
        name="Charlie",
        email="charlie@example.com",
        age=35
    )
except ValidationError as e:
    print(e.json())
    # Affiche les erreurs de validation au format JSON
    # avec le champ, le type d'erreur et le message

Field - Paramètres avancés

Field permet d'ajouter des contraintes et de la métadonnée aux champs d'un modèle.

Contraintes de validation

from pydantic import BaseModel, Field

class Product(BaseModel):
    name: str = Field(min_length=3, max_length=50)
    description: Optional[str] = Field(None, max_length=500)
    price: float = Field(gt=0, le=10000)  # gt = greater than, le = less than or equal
    quantity: int = Field(default=0, ge=0)  # ge = greater than or equal
    tags: List[str] = Field(default_factory=list, max_length=10)
    
# Exemple d'utilisation
product = Product(
    name="Laptop",
    description="Powerful laptop for developers",
    price=1299.99,
    quantity=5,
    tags=["electronics", "computers"]
)

Valeurs par défaut et alias

from pydantic import BaseModel, Field

class Article(BaseModel):
    title: str = Field(..., description="Titre de l'article")
    content: str = Field(..., min_length=10)

    views: int = Field(default=0, ge=0)
    published_at: Optional[datetime] = None

# Utilisation avec alias
article = Article(
    title="Introduction à Pydantic",
    content="Contenu de l'article...",
    authorName="Alice"  # Utilise l'alias
)

print(article.author_name)  # Alice

ConfigDict - Configuration du modèle

ConfigDict permet de configurer le comportement global d'un modèle Pydantic.

Configuration de base

from pydantic import BaseModel, ConfigDict, Field

class Person(BaseModel):
    model_config = ConfigDict(
        str_strip_whitespace=True,  # Supprime les espaces en début/fin
        validate_assignment=True,    # Valide lors de l'assignation
        use_enum_values=True,        # Utilise les valeurs d'enum
        strict=False                 # Mode strict désactivé (permet conversion)
    )
    
    name: str
    email: str

# str_strip_whitespace en action
person = Person(name="  Alice  ", email="  alice@example.com  ")
print(person.name)  # "Alice" (espaces supprimés)

validate_assignment

class User(BaseModel):
    model_config = ConfigDict(validate_assignment=True)
    
    username: str = Field(min_length=3)
    age: int = Field(ge=0, le=150)

user = User(username="alice", age=25)

# Validation lors de l'assignation
try:
    user.username = "ab"  # Trop court
except ValidationError as e:
    print("Erreur:", e)

# Sans validate_assignment=True, cette assignation
# passerait sans validation

Validators personnalisés

Pydantic permet de créer des validators personnalisés avec le décorateur @field_validator.

field_validator de base

from pydantic import BaseModel, field_validator

class User(BaseModel):
    username: str
    email: str
    age: int
    
    @field_validator('username')
    @classmethod
    def username_alphanumeric(cls, v):
        if not v.isalnum():
            raise ValueError('Le username doit être alphanumérique')
        return v
    
    @field_validator('email')
    @classmethod
    def email_must_contain_at(cls, v):
        if '@' not in v:
            raise ValueError('Email invalide')
        return v.lower()  # Normalise l'email en minuscules
    
    @field_validator('age')
    @classmethod
    def age_must_be_valid(cls, v):
        if v < 18:
            raise ValueError('Doit avoir au moins 18 ans')
        if v > 120:
            raise ValueError('Âge invalide')
        return v

# Utilisation
user = User(username="alice123", email="ALICE@example.com", age=25)
print(user.email)  # alice@example.com (converti en minuscules)

Validators avec plusieurs champs

from pydantic import BaseModel, field_validator, model_validator

class DateRange(BaseModel):

        return self

# Utilisation

Modes de validation : before vs after

Les validators peuvent s'exécuter à différents moments avec le paramètre mode.

Mode 'before' - Avant la validation de type

Le validator s'exécute avant la conversion de type Pydantic. Utile pour transformer les données brutes avant validation.

from pydantic import BaseModel, field_validator

class DataProcessor(BaseModel):
    raw_data: str
    
    @field_validator('raw_data', mode='before')
    @classmethod
    def preprocess_data(cls, v):
        # Exécuté AVANT la validation str
        # Reçoit les données brutes (peut être bytes, int, etc.)
        
        if isinstance(v, bytes):
            return v.decode('utf-8')  # Convertir bytes en str
        if isinstance(v, int):
            return str(v)  # Convertir int en str
        return v

# Exemples
processor1 = DataProcessor(raw_data=b"hello")  # bytes converti
processor2 = DataProcessor(raw_data=123)       # int converti
processor3 = DataProcessor(raw_data="text")    # str accepté
print(processor1.raw_data)  # "hello"

Mode 'after' - Après la validation de type

Le validator s'exécute après la conversion de type. Les données ont déjà le bon type Python.

class DataProcessor(BaseModel):
    raw_data: str
    
    @field_validator('raw_data', mode='after')
    @classmethod
    def postprocess_data(cls, v):
        # Exécuté APRÈS la validation str
        # v est garanti être un str à ce stade
        return v.strip().upper()  # Nettoyer et mettre en majuscules

processor = DataProcessor(raw_data="  hello world  ")
print(processor.raw_data)  # "HELLO WORLD"

Combinaison before et after

class EmailProcessor(BaseModel):
    email: str
    
    @field_validator('email', mode='before')
    @classmethod
    def convert_to_string(cls, v):
        """Convertir en string si nécessaire"""
        if not isinstance(v, str):
            return str(v)
        return v
    
    @field_validator('email', mode='after')
    @classmethod
    def normalize_email(cls, v):
        """Normaliser l'email (après conversion en str)"""
        v = v.strip().lower()
        if '@' not in v:
            raise ValueError('Email invalide')
        return v

# Les deux validators s'exécutent dans l'ordre
email = EmailProcessor(email="  USER@EXAMPLE.COM  ")
print(email.email)  # "user@example.com"

Comparaison des modes

from typing import Any

class ComparisonExample(BaseModel):
    value: int
    
    @field_validator('value', mode='before')
    @classmethod
    def before_validator(cls, v: Any):
        print(f"BEFORE - Type reçu: {type(v).__name__}, Valeur: {v}")
        # Peut recevoir n'importe quel type
        return v
    
    @field_validator('value', mode='after')
    @classmethod
    def after_validator(cls, v: int):
        print(f"AFTER - Type reçu: {type(v).__name__}, Valeur: {v}")
        # Reçoit toujours un int
        return v

# Test
example = ComparisonExample(value="42")
# Output:
# BEFORE - Type reçu: str, Valeur: 42
# AFTER - Type reçu: int, Valeur: 42

model_validator - Validation au niveau du modèle

Le décorateur @model_validator permet de valider l'ensemble du modèle, pas seulement un champ. Il a 3 modes : before, after et wrap.

Mode 'before' - Validation avant création

S'exécute avant la validation de tous les champs. Reçoit les données brutes sous forme de dictionnaire.

from pydantic import BaseModel, model_validator
from typing import Dict, Any

class User(BaseModel):
    username: str
    email: str
    password: str
    
    @model_validator(mode='before')
    @classmethod
    def check_raw_data(cls, data: Dict[str, Any]) -> Dict[str, Any]:
        """Valide et transforme les données brutes"""
        print(f"Mode 'before' - données reçues: {type(data)}")
        
        # Vérifier que les champs requis sont présents
        if 'username' not in data and 'email' in data:
            # Générer username depuis email
            data['username'] = data['email'].split('@')[0]
        
        # Transformer les données
        if 'email' in data:
            data['email'] = data['email'].lower()
        
        # Ajouter des valeurs par défaut
        if 'password' not in data:
            data['password'] = 'default_password'
        
        return data

# Username généré automatiquement
user = User(email="Alice@Example.com", password="secret123")
print(user.username)  # "alice"
print(user.email)     # "alice@example.com"

Mode 'after' - Validation après création

S'exécute après la validation de tous les champs. Reçoit l'instance du modèle complète.

from pydantic import BaseModel, model_validator
from datetime import datetime

class DateRange(BaseModel):
    start_date: datetime
    end_date: datetime
    description: str = ""
    
    @model_validator(mode='after')
    def check_dates_consistency(self):
        """Valide la cohérence entre plusieurs champs"""
        print(f"Mode 'after' - données reçues: {type(self)}")
        
        # Accès direct aux attributs validés
        if self.start_date >= self.end_date:
            raise ValueError(
                f"start_date ({self.start_date}) doit être avant end_date ({self.end_date})"
            )
        
        # Calculer la durée
        duration = (self.end_date - self.start_date).days
        if duration > 365:
            raise ValueError(f"La période ne peut pas dépasser 1 an (actuellement {duration} jours)")
        
        # Peut modifier les attributs
        if not self.description:
            self.description = f"Période de {duration} jours"
        
        return self

# Validation réussie
date_range = DateRange(
    start_date=datetime(2026, 1, 1),
    end_date=datetime(2026, 12, 31)
)
print(date_range.description)  # "Période de 364 jours"

Mode 'wrap' - Contrôle total de la validation

Le mode le plus puissant : permet de contrôler entièrement le processus de validation.

from pydantic import BaseModel, model_validator, ValidationError
from pydantic_core import ValidationInfo
from typing import Any
import logging

logger = logging.getLogger(__name__)

class SecureData(BaseModel):
    username: str
    password: str
    api_key: str
    
    @model_validator(mode='wrap')
    @classmethod
    def validate_with_logging(cls, data: Any, handler, info: ValidationInfo):
        """
        Mode 'wrap' - Enveloppe la validation standard
        - data: données brutes
        - handler: fonction de validation Pydantic
        - info: informations de contexte
        """
        print(f"Mode 'wrap' - Type données: {type(data)}")
        
        # AVANT la validation standard
        logger.info(f"Tentative de validation pour: {info.field_name}")
        
        try:
            # Appeler la validation standard de Pydantic
            validated = handler(data)
            
            # APRÈS la validation standard (si succès)
            logger.info("Validation réussie")
            
            # Masquer les données sensibles dans les logs
            logger.info(f"User créé: {validated.username}")
            
            return validated
            
        except ValidationError as e:
            # Gérer les erreurs de validation
            logger.error(f"Erreur de validation: {e}")
            # Peut modifier l'erreur ou tenter une récupération
            raise
        except Exception as e:
            # Gérer d'autres erreurs
            logger.error(f"Erreur inattendue: {e}")
            raise

# Utilisation
secure = SecureData(
    username="alice",
    password="secret123",
    api_key="key_abc123"
)

Cas d'usage avancé : Validation conditionnelle

from pydantic import BaseModel, model_validator, Field
from typing import Optional

class Product(BaseModel):
    name: str
    price: float = Field(gt=0)
    discount_percentage: Optional[float] = Field(None, ge=0, le=100)
    final_price: Optional[float] = None
    is_available: bool = True
    stock: int = Field(ge=0)
    
    @model_validator(mode='after')
    def calculate_and_validate(self):
        """Validation et calculs complexes"""
        
        # Calculer le prix final si discount présent
        if self.discount_percentage:
            discount_amount = self.price * (self.discount_percentage / 100)
            self.final_price = self.price - discount_amount
        else:
            self.final_price = self.price
        
        # Validation conditionnelle
        if not self.is_available and self.stock > 0:
            raise ValueError("Un produit avec du stock ne peut pas être indisponible")
        
        if self.is_available and self.stock == 0:
            # Auto-correction
            self.is_available = False
        
        # Prix minimum
        if self.final_price < 0.01:
            raise ValueError(f"Prix final trop bas: {self.final_price}")
        
        return self

# Exemple avec discount
product1 = Product(
    name="Laptop",
    price=1000,
    discount_percentage=20,
    stock=5
)
print(f"Prix final: {product1.final_price}")  # 800.0

# Auto-correction du stock
product2 = Product(
    name="Mouse",
    price=25,
    is_available=True,
    stock=0  # Stock vide, is_available sera mis à False
)
print(f"Disponible: {product2.is_available}")  # False

Combinaison de model_validator et field_validator

class Order(BaseModel):
    order_id: str
    customer_email: str
    items: list[str]
    total_amount: float
    tax_amount: Optional[float] = None
    final_amount: Optional[float] = None
    
    # Field validator s'exécute en premier
    @field_validator('customer_email')
    @classmethod
    def validate_email(cls, v):
        if '@' not in v:
            raise ValueError('Email invalide')
        return v.lower()
    
    @field_validator('items')
    @classmethod
    def validate_items(cls, v):
        if len(v) == 0:
            raise ValueError('La commande doit contenir au moins un article')
        return v
    
    # Model validator (before) s'exécute avant les field validators
    @model_validator(mode='before')
    @classmethod
    def preprocess(cls, data):
        # Générer order_id si absent
        if 'order_id' not in data:
            import uuid
            data['order_id'] = f"ORD-{uuid.uuid4().hex[:8]}"
        return data
    
    # Model validator (after) s'exécute après tous les field validators
    @model_validator(mode='after')
    def calculate_totals(self):
        # Calculer la taxe (20%)
        self.tax_amount = self.total_amount * 0.20
        self.final_amount = self.total_amount + self.tax_amount
        return self

# Ordre d'exécution:
# 1. model_validator(mode='before')
# 2. field_validator pour chaque champ
# 3. model_validator(mode='after')

order = Order(
    customer_email="ALICE@EXAMPLE.COM",
    items=["item1", "item2"],
    total_amount=100
)
print(f"Order ID: {order.order_id}")        # Généré automatiquement
print(f"Email: {order.customer_email}")     # "alice@example.com" (normalisé)
print(f"Final: {order.final_amount}")       # 120.0 (avec taxe)

@classmethod et @staticmethod

Utilisation de méthodes de classe et statiques avec Pydantic pour des fonctionnalités avancées.

@classmethod pour les constructeurs alternatifs

from pydantic import BaseModel
from datetime import datetime

class Event(BaseModel):
    title: str
    description: str
    event_date: datetime
    created_at: datetime
    
    @classmethod
    def create_now(cls, title: str, description: str, days_ahead: int = 0):
        """Constructeur alternatif pour créer un événement"""
        from datetime import timedelta
        now = datetime.now()
        event_date = now + timedelta(days=days_ahead)
        
        return cls(
            title=title,
            description=description,
            event_date=event_date,
            created_at=now
        )
    
    @classmethod
    def from_dict(cls, data: dict):
        """Créer un événement depuis un dictionnaire avec transformation"""
        return cls(
            title=data.get('title', 'Sans titre'),
            description=data.get('desc', ''),
            event_date=datetime.fromisoformat(data['date']),
            created_at=datetime.now()
        )

# Utilisation
event1 = Event.create_now("Réunion", "Réunion hebdomadaire", days_ahead=7)
event2 = Event.from_dict({
    'title': 'Conference',
    'desc': 'Tech conference',
    'date': '2026-06-15T10:00:00'
})

@staticmethod pour les utilitaires

class Coordinates(BaseModel):
    latitude: float = Field(ge=-90, le=90)
    longitude: float = Field(ge=-180, le=180)
    
    @staticmethod
    def is_valid_latitude(lat: float) -> bool:
        """Vérifie si une latitude est valide"""
        return -90 <= lat <= 90
    
    @staticmethod
    def is_valid_longitude(lon: float) -> bool:
        """Vérifie si une longitude est valide"""
        return -180 <= lon <= 180
    
    @staticmethod
    def calculate_distance(coord1: 'Coordinates', coord2: 'Coordinates') -> float:
        """Calcule la distance entre deux points (formule simplifiée)"""
        import math
        lat1, lon1 = math.radians(coord1.latitude), math.radians(coord1.longitude)
        lat2, lon2 = math.radians(coord2.latitude), math.radians(coord2.longitude)
        
        dlat = lat2 - lat1
        dlon = lon2 - lon1
        
        a = math.sin(dlat/2)**2 + math.cos(lat1) * math.cos(lat2) * math.sin(dlon/2)**2
        c = 2 * math.asin(math.sqrt(a))
        
        # Rayon de la Terre en km
        r = 6371
        return c * r

# Utilisation
if Coordinates.is_valid_latitude(48.8566):
    paris = Coordinates(latitude=48.8566, longitude=2.3522)

london = Coordinates(latitude=51.5074, longitude=-0.1278)
distance = Coordinates.calculate_distance(paris, london)
print(f"Distance: {distance:.2f} km")

Pydantic Dataclasses

Pydantic propose un décorateur @dataclass qui combine les dataclasses Python avec la validation Pydantic.

Dataclass de base

from pydantic.dataclasses import dataclass
from pydantic import Field

@dataclass
class Point:
    x: float
    y: float
    label: str = "Origin"

# Utilisation identique aux dataclasses
point = Point(x=10.5, y=20.3, label="A")
print(point)  # Point(x=10.5, y=20.3, label='A')

Dataclass avec validation

from pydantic.dataclasses import dataclass
from pydantic import Field, field_validator
from typing import List

@dataclass
class Rectangle:
    width: float = Field(gt=0)
    height: float = Field(gt=0)
    color: str = "white"
    
    @field_validator('color')
    @classmethod
    def validate_color(cls, v):
        allowed = ['white', 'black', 'red', 'green', 'blue']
        if v not in allowed:
            raise ValueError(f'Couleur doit être parmi {allowed}')
        return v
    
    @property
    def area(self) -> float:
        return self.width * self.height
    
    @property
    def perimeter(self) -> float:
        return 2 * (self.width + self.height)

# Utilisation
rect = Rectangle(width=10, height=5, color="blue")
print(f"Area: {rect.area}")  # Area: 50.0
print(f"Perimeter: {rect.perimeter}")  # Perimeter: 30.0

Modèles de données OpenMeteo

from pydantic import BaseModel, Field, field_validator, ConfigDict
from typing import List, Optional
from datetime import datetime

class CurrentWeather(BaseModel):
    """Données météo actuelles"""
    model_config = ConfigDict(str_strip_whitespace=True)
    
    temperature: float = Field(alias="temperature_2m")
    windspeed: float = Field(alias="wind_speed_10m", ge=0)
    winddirection: int = Field(alias="wind_direction_10m", ge=0, le=360)
    weathercode: int = Field(alias="weather_code")
    time: datetime
    
    @field_validator('temperature')
    @classmethod
    def check_temperature_range(cls, v):
        if v < -100 or v > 60:
            raise ValueError('Température invalide (doit être entre -100 et 60°C)')
        return v
    
    @classmethod
    def get_weather_description(cls, code: int) -> str:
        """Retourne la description météo selon le code"""
        weather_codes = {
            0: "Ciel dégagé",
            1: "Principalement dégagé",
            2: "Partiellement nuageux",
            3: "Couvert",
            45: "Brouillard",
            48: "Brouillard givrant",
            51: "Bruine légère",
            61: "Pluie légère",
            63: "Pluie modérée",
            65: "Pluie forte",
            71: "Neige légère",
            95: "Orage"
        }
        return weather_codes.get(code, "Inconnu")

class HourlyData(BaseModel):
    """Données horaires"""
    time: List[datetime]
    temperature_2m: List[float]
    precipitation_probability: List[int] = Field(alias="precipitation_probability")
    
    @field_validator('precipitation_probability')
    @classmethod
    def validate_probabilities(cls, v):
        for prob in v:
            if not 0 <= prob <= 100:
                raise ValueError('La probabilité doit être entre 0 et 100')
        return v
    
    @staticmethod
    def get_avg_temperature(temperatures: List[float]) -> float:
        """Calcule la température moyenne"""
        return sum(temperatures) / len(temperatures) if temperatures else 0.0

class DailyData(BaseModel):
    """Données journalières"""
    time: List[datetime]
    temperature_2m_max: List[float] = Field(alias="temperature_2m_max")
    temperature_2m_min: List[float] = Field(alias="temperature_2m_min")
    sunrise: List[datetime]
    sunset: List[datetime]
    
    @model_validator(mode='after')
    def validate_min_max(self):
        for tmin, tmax in zip(self.temperature_2m_min, self.temperature_2m_max):
            if tmin > tmax:
                raise ValueError('Température min ne peut être > température max')
        return self

class OpenMeteoResponse(BaseModel):
    """Réponse complète de l'API OpenMeteo"""
    model_config = ConfigDict(
        str_strip_whitespace=True,
        json_schema_extra={
            "examples": [{
                "latitude": 48.8566,
                "longitude": 2.3522,
                "timezone": "Europe/Paris",
                "current": {
                    "temperature_2m": 15.5,
                    "wind_speed_10m": 10.2,
                    "wind_direction_10m": 180,
                    "weather_code": 1,
                    "time": "2026-01-12T14:30:00"
                }
            }]
        }
    )
    
    latitude: float = Field(ge=-90, le=90)
    longitude: float = Field(ge=-180, le=180)
    timezone: str
    timezone_abbreviation: str
    elevation: float
    current: Optional[CurrentWeather] = None
    hourly: Optional[HourlyData] = None
    daily: Optional[DailyData] = None
    
    @field_validator('timezone')
    @classmethod
    def validate_timezone(cls, v):
        # Validation simple du format timezone
        if '/' not in v:
            raise ValueError('Format timezone invalide (attendu: Continent/Ville)')
        return v
    
    @classmethod
    def from_api(cls, data: dict):
        """Constructeur depuis les données brutes de l'API"""
        return cls(**data)
    
    def get_current_temp_celsius(self) -> Optional[float]:
        """Retourne la température actuelle en Celsius"""
        return self.current.temperature if self.current else None
    
    def get_weather_summary(self) -> str:
        """Retourne un résumé météo"""
        if not self.current:
            return "Pas de données actuelles"
        
        desc = CurrentWeather.get_weather_description(self.current.weathercode)
        temp = self.current.temperature
        wind = self.current.windspeed
        
        return f"{desc}, {temp}°C, vent {wind} km/h"

import requests
from pydantic import ValidationError

def fetch_weather(latitude: float, longitude: float):
    """Récupère et valide les données météo"""
    
    url = "https://api.open-meteo.com/v1/forecast"
    params = {
        "latitude": latitude,
        "longitude": longitude,
        "current": ["temperature_2m", "wind_speed_10m", "wind_direction_10m", "weather_code"],
        "hourly": ["temperature_2m", "precipitation_probability"],
        "daily": ["temperature_2m_max", "temperature_2m_min", "sunrise", "sunset"],
        "timezone": "auto"
    }
    
    try:
        # Requête API
        response = requests.get(url, params=params)
        response.raise_for_status()
        data = response.json()
        
        # Validation avec Pydantic
        weather = OpenMeteoResponse.from_api(data)
        
        print(f"📍 Position: {weather.latitude}, {weather.longitude}")
        print(f"🌍 Timezone: {weather.timezone}")
        print(f"📊 Altitude: {weather.elevation}m")
        
        if weather.current:
            print(f"\n🌤️  Météo actuelle:")
            print(f"   {weather.get_weather_summary()}")
            print(f"   Direction du vent: {weather.current.winddirection}°")
        
        if weather.hourly:
            avg_temp = HourlyData.get_avg_temperature(weather.hourly.temperature_2m[:24])
            print(f"\n📈 Température moyenne (24h): {avg_temp:.1f}°C")
        
        if weather.daily:
            print(f"\n📅 Prévisions journalières:")
            for i, date in enumerate(weather.daily.time[:3]):
                tmin = weather.daily.temperature_2m_min[i]
                tmax = weather.daily.temperature_2m_max[i]
                print(f"   {date.strftime('%d/%m')}: {tmin}°C - {tmax}°C")
        
        return weather
        
    except requests.RequestException as e:
        print(f"❌ Erreur API: {e}")
        return None
        
    except ValidationError as e:
        print(f"❌ Erreur de validation:")
        for error in e.errors():
            field = " -> ".join(str(loc) for loc in error['loc'])
            print(f"   • {field}: {error['msg']}")
        return None
        
    except Exception as e:
        print(f"❌ Erreur inattendue: {e}")
        return None

def validate_email_format(email: str) -> str:
    if '@' not in email or '.' not in email:
        raise ValueError('Format email invalide')
    return email.lower()

class User(BaseModel):
    email: str
    
    @field_validator('email')
    @classmethod
    def check_email(cls, v):
        return validate_email_format(v)
